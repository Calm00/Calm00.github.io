# 矩阵 (Matrix) 定义

一个 n×mn\times mn×m 的矩阵是一个由 nnn 行 mmm 列元素排成的矩形阵列。矩阵里的元素可以是数字符号或者数学式。

形如(abcd)\left(\begin{matrix}a&b\\c&d\end{matrix}\right)(ac​bd​) 的数表称为二阶矩阵，它由二行二列组成，其中a,b,c,da,b,c,da,b,c,d 称为这个矩阵的元素。

形如(x1x2)\left(\begin{matrix}x_1\\x_2\end{matrix}\right)(x1​x2​​) 的有序对称为列向量 (Column Vector)

# 方法一

二阶矩阵的乘法满足结合律

设A,B,CA,B,CA,B,C 都是任意的二阶矩阵

则A(BC)=(AB)CA(BC)=(AB)CA(BC)=(AB)C

不在此证明

设n=N÷2n=N\div 2n=N÷2 (结果向下取整)

若 N∈2kN\in 2kN∈2k 则 AN=An×AnA^N=A^n\times A^nAN=An×An

若 N∈2k+1N\in 2k+1N∈2k+1 则 AN=An×An×AA^N=A^n\times A^n\times AAN=An×An×A

这样可以减少计算次数，自行思考原因

# 方法二

以计算 A6A^6A6 为例

例如6_{ (10) }=110_

则 A6=A4×A2A^6=A^4\times A^2A6=A4×A2

int qpow(int a, int k) {
    int res = 1;
    while(k) {
        if(a & 1) res *= a;
        k >>= 1;
        a *= a;
    }
    return res;
}

#include<cstdio>
#include<cstring>

struct Matrix {
	long long fib[2][2];
	Matrix() {memset(fib, 0, sizeof fib);}
};

Matrix qmul(Matrix &a, Matrix &b) {
	Matrix c;
	for(int i = 0; i < 2; ++i) {
		for(int j = 0; j < 2; ++j) {
			for(int k = 0; k < 2; ++k) {
				c.fib[i][j] += a.fib[i][k] * b.fib[k][j];
			}
		}
	}
	return c;
}

Matrix qpow(Matrix a, long long k) {
	Matrix ans;
	ans.fib[0][0] = ans.fib[1][1] = 1;
	while(k) {
		if(k & 1) ans = qmul(ans, a);
		a = qmul(a, a);
		k >>= 1;
	}
	return ans;
}

int main() {
	long long n;
	Matrix a, b;
	scanf("%lld", &n);
	a.fib[0][0] = a.fib[0][1] = a.fib[1][0] = 1;
	b = qpow(a, n);
	printf("%lld\n", b.fib[0][1]);
	return 0;
}
