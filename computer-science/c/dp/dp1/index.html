# 引例 1：斐波那契数列 Fibonacci

斐波那契数列：1,1,2,3,5,8,13,21.......

通项公式：fn=fn−2+fn−1f_n=f_{n-2}+f_{n-1}fn​=fn−2​+fn−1​ (n>2)，其中f1=f2=1f_1=f_2=1f1​=f2​=1

输入：n∈Z+(n≤1000)n\in Z_+(n\leq 1000)n∈Z+​(n≤1000)

输出：fnf_nfn​ mod 10000

大家可以自己思考一下

# 方法一

建议自行思考一段时间哦~（点此查看代码）
#include<cstdio>

int n;

int dp(int x) {
    if(x == 1 || x == 2) return 1;
    return (dp(x - 2) + dp(x - 1)) % 10000;
}

int main() {
    scanf("%d", &n);
    printf("%d\n", dp(n));
    return 0;
}

# 改进一（记忆化搜索）

#include<cstdio>

int n, f[1005];

int dp(int x) {
	if(f[x]) return f[x];
    if(x == 1 || x == 2) return 1;
    return f[x] = (dp(x - 2) + dp(x - 1)) % 10000;
}

int main() {
    scanf("%d", &n);
    printf("%d\n", dp(n));
    return 0;
}

# 改进二（直接递推）

此方法比较常用

#include<cstdio>

int n, f[1005];

int main() {
    scanf("%d", &n);
    f[1] = f[2] = 1;
    for(int i = 3; i <= n; ++i) 
    	f[i] = (f[i - 1] + f[i - 2]) % 10000;
    printf("%d\n", f[n]);
    return 0;
}
